# sample

## LogAgent

```

func ReadLogWithTailf() {
	ch := make(chan bool)
	for _, LogpathStruct := range models.EtcdValue {
		models.NewTailf(LogpathStruct.LogPath)
		go LogRun(models.Tails, LogpathStruct.Topic, ch)
	}

	for v := range ch {
		fmt.Println(v)
	}
}

```
